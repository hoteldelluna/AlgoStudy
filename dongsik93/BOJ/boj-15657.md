`code`

```python
def f(index, n, m, k):
    if(index  == m):
        print(*res)
        return
    else:
        for i in range(n):
            if(k <= arr[i]):
                visited[i] = 1
                res[index] = arr[i]
                k = arr[i]
                f(index+1, n, m, k)
                visited[i] = 0 


n, m = map(int, input().split())
arr = sorted(list(map(int, input().split())))
visited = [0 for i in range(n)]
res = [0] * m
f(0,n,m,0)
```
